Output

Additional options

With value

<input type="text" class="datepicker" name="date" value="2017-04-14">

With min-date

<input type="text" class="datepicker" name="date" data-min-date="today">

<input type="text" class="datepicker" name="date" value="tomorrow" data-min-date="tomorrow">

<input type="text" class="datepicker" name="date" value="2020-10-10">

With max-date

<input type="text" class="datepicker" name="date" data-max-date="yesterday">

With min-date and max-date

<input type="text" class="datepicker" name="date" value="today" data-min-date="yesterday" data-max-date="tomorrow">

<input type="text" class="datepicker" name="date" value="2018-06-22" data-min-date="2018-03-22" data-max-date="2018-09-22">

With time

<input type="text" class="js-datepicker" name="date" data-format="datetime" >
<input type="text" class="js-datepicker" name="date" data-format="datetime" value="2018-05-25 21:08:00">
